    public void slctPaysSport() {
        
        // si les deux liste on une ligne de selectioné
        if(lstPays.getSelectedIndex() > -1 && lstSports.getSelectedIndex() > -1) {
          modeleAthletes.loadData((Pays)modelePays.get(), (Sport)modeleSports.get());
        }
    }
